from PC

select distinct maker
from Product
where type = 'Printer' -- 2(1) ok

select model, ram, screen
from Laptop
where price > 1000 -- 3(1) ok

select *
from Printer
where color = 'y' -- 4(1) ok

from PC
where (cd = '12x' or cd = '24x') and price < 600 -- 5(1) ok
